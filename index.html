// 🎯 محرك Ziko Sniper V12: كشف التجميع قبل الانفجار async function checkLiquidity() { openModal('📡 رادار التجميع الذكي', '
جاري تحليل بصمات "السيولة الذكية"... ⏳
', '🎯'); const data = await smartFetch('https://api.binance.com/api/v3/ticker/24hr'); if(!data) return; // خوارزمية المحترفين: // 1. سيولة حقيقية (Volume > $20M) لاستبعاد العملات الضعيفة. // 2. تذبذب سعر ضيق (0.2% < Change < 3.2%) لضمان أن الانفجار لم يحدث بعد. // 3. معامل القوة (Intensity): نسبة السيولة إلى التغير السعري. const prePumpCoins = data .filter(i => { const vol = parseFloat(i.quoteVolume); const change = parseFloat(i.priceChangePercent); return i.symbol.endsWith('USDT') && vol > 20000000 && change >= 0.2 && change <= 3.2; }) .sort((a, b) => parseFloat(b.quoteVolume) - parseFloat(a.quoteVolume)) .slice(0, 4); if (prePumpCoins.length === 0) { openModal('رادار القنص', '
البحث جاري... السيولة الحالية منتشرة بشكل طبيعي ولا يوجد تجميع مريب حالياً.
', '🔍'); return; } let html = `

تحليل المهندس: تم رصد "ضغط سيولة" عالي مع استقرار سعري نسبي.

`; prePumpCoins.forEach(coin => { const name = coin.symbol.replace('USDT', ''); const vol = (parseFloat(coin.quoteVolume) / 1e6).toFixed(1); const price = parseFloat(coin.lastPrice); const change = parseFloat(coin.priceChangePercent).toFixed(2); // حساب معامل الشحن (Charge Level) const chargeLevel = Math.min(98, (vol / 10) + 40).toFixed(0); html += `
ASSET

${name}

CURRENT PRICE
$${price}
تدفق السيولة (24h) $${vol}M
التغير اللحظي +${change}%
مستوى الشحن (Accumulation) ${chargeLevel}%
`; }); html += `
`; openModal('رادار السيولة الذكية', html, '📡'); }
